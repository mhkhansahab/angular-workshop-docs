## Create Signal for categories

`categories` contains all categories.\
`selectedCategory` is a signal that contains the currently selected category.
`getProductsByCategory` is a method that will filter the products based on the selected category.
`getSelectedCategory` is a method that will return the currently selected category but as a readonly signal.

<Tabs>
  <TabItem label="home.component.ts">
  ```ts
  hlconfig 2,10,11,14-25
import { Injectable, signal } from '@angular/core';
import { products, categories /* ğŸ‘ˆğŸ½ add this */ } from '../../static/mock-data';
import { IProduct } from '../interfaces/product.interface';

@Injectable({
  providedIn: 'root',
})

export class ProductService {
  categories: string[] = categories; /* ğŸ‘ˆğŸ½ add this */
  private selectedCategory = signal<string>('all'); /* ğŸ‘ˆğŸ½ add this */
  private selectedProducts = signal<IProduct[]>(products);

  getProductsByCategory(category: string) { /* ğŸ‘ˆğŸ½ add this */
    const filteredProducts =
      category === 'all'
        ? products
        : products.filter((product) => product.category === category);
    this.selectedCategory.set(category);
    this.selectedProducts.set(filteredProducts);
  }

  getSelectedCategory() { /* ğŸ‘ˆğŸ½ add this */
    return this.selectedCategory.asReadonly();
  }

  getSelectedProducts() {
      return this.selectedProducts.asReadonly();
  }
}
```

## Listening to signal inside component

<Tabs>
<TabItem label="src/app/app.component.ts">
```ts
hlconfig 13-21
import { Component, inject } from '@angular/core';
import { ProductService } from './shared/services/product.service';

@Component({
selector: 'app-root',
templateUrl: './app.component.html',
styleUrl: './app.component.scss'
})
export class AppComponent {
  productService = inject(ProductService);

  selectedProducts = this.productService.getSelectedProducts();
  selectedCategory = this.productService.getSelectedCategory(); /* ğŸ‘ˆğŸ½ add this */

  getProductsByCategory(category: string) { /* ğŸ‘ˆğŸ½ add this */
    this.productService.getProductsByCategory(category);
  }

  isSelected(category: string) { /* ğŸ‘ˆğŸ½ add this */
    return category === this.selectedCategory();
  }
}
````
